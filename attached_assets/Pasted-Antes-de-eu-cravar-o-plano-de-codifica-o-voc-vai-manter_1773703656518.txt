=== QUEBRA DE PÁGINA ===

1. Visão Executiva
1.1 O que é o BeUpFree (definição oficial)
O BeUpFree é uma plataforma afiliada de curadoria e publicação de ofertas esportivas, com foco inicial em tênis, que:

coleta ofertas em marketplaces parceiros (ex.: Mercado Livre, Shopee, Amazon, Netshoes, Centauro)
trata e normaliza dados (produto ≠ oferta)
passa por curadoria (triagem humana)
publica um catálogo útil e navegável
monetiza via cliques em links de afiliado (não vende diretamente)
1.2 Problema que resolve
Marketplaces são ótimos para comprar e péssimos para “descobrir rápido o que vale a pena” quando você quer:

navegar por intenção (tênis corrida barato, nike em promoção etc.)
filtrar por critérios práticos
ver ofertas reais (com menos ruído)
1.3 Proposta de valor

Menos ruído, mais intenção.
Catálogo organizado (marca/categoria/uso/faixa de preço).
Páginas SEO com recortes que fazem sentido.
Curadoria como filtro de qualidade.
1.4 Métrica norte (North Star Metric)
Cliques qualificados em ofertas publicadas (affiliate clicks) por sessão/página, com CTR sustentável.

3. Requisitos
3.1 Requisitos Funcionais (RF)
RF-01 — Coletar ofertas
O sistema deve coletar ofertas de marketplaces configurados e persistir os itens brutos com rastreabilidade.

RF-02 — Armazenar item bruto (staging)
Todo item coletado deve ser armazenado sem perda do payload original.

RF-03 — Processar itens
O sistema deve normalizar título, sugerir marca/categoria, detectar duplicidade e gerar item processado.

RF-04 — Fila de triagem
O sistema deve manter uma fila com status e prioridade para curadoria.

RF-05 — Curadoria
Operador deve aprovar/rejeitar e registrar justificativa/ação.

RF-06 — Publicação
Aprovação deve gerar/atualizar produto e criar/atualizar oferta publicada.

RF-07 — Catálogo público
Usuário deve listar ofertas publicadas com filtros (mínimo: marca, categoria, faixa de preço).

RF-08 — Página de produto
Usuário deve ver detalhes do produto e ofertas associadas (no mínimo: melhor oferta).

RF-09 — Tracking de cliques
Ao clicar numa oferta, registrar evento e redirecionar para URL destino (afiliada).

RF-10 — SEO pages
Sistema deve suportar páginas por intenção (marca/categoria/preço/uso), com slugs estáveis.

3.2 Requisitos Não Funcionais (RNF)
RNF-01 — Separação público/admin
Rotas e permissões administrativas não podem se misturar com rotas públicas.

RNF-02 — Segurança de dados
Service Role Key do Supabase nunca pode ir ao frontend. Operações sensíveis devem acontecer no backend.

RNF-03 — Observabilidade
Coleta/processamento/publicação devem gerar logs auditáveis.

RNF-04 — Desempenho
Listagens públicas devem responder rápido e suportar cache/paginação.

RNF-05 — Evolução
Modelo deve permitir múltiplas ofertas por produto, várias fontes e expansão para novas categorias.

import { supabaseAdmin } from "../lib/supabaseAdmin";
import { sha1 } from "../lib/hash";

type RawOfferInput = {
  marketplaceId: number;
  externalItemId?: string;
  title?: string;
  price?: number;
  oldPrice?: number;
  itemUrl?: string;
  imageUrl?: string;
  payload: any;
};

export async function runCollect(params: {
  collectionSourceId: number;
  executionType: "search" | "refresh" | "import";
  searchTerm?: string;
  items: RawOfferInput[]; // no MVP você injeta aqui o retorno do seu scraper/serviço
}) {
  const { data: batch, error: batchErr } = await supabaseAdmin
    .from("collection_batches")
    .insert({
      collection_source_id: params.collectionSourceId,
      execution_type: params.executionType,
      search_term: params.searchTerm ?? null,
      batch_status: "running",
    })
    .select("*")
    .single();

  if (batchErr) throw batchErr;

  const rows = params.items.map((it) => {
    const dedupHash = it.externalItemId
      ? sha1(`${it.marketplaceId}:${it.externalItemId}`)
      : sha1(`${it.marketplaceId}:${(it.title ?? "").toLowerCase()}:${it.price ?? ""}:${it.imageUrl ?? ""}`);

    return {
      collection_batch_id: batch.id,
      marketplace_id: it.marketplaceId,
      external_item_id: it.externalItemId ?? null,
      raw_title: it.title ?? null,
      raw_price: it.price ?? null,
      raw_old_price: it.oldPrice ?? null,
      item_url: it.itemUrl ?? null,
      image_url: it.imageUrl ?? null,
      payload_json: it.payload,
      dedup_hash: dedupHash,
      processing_status: "pending",
    };
  });

  const { error: insertErr } = await supabaseAdmin
    .from("raw_collected_items")
    .insert(rows);

  if (insertErr) throw insertErr;

  await supabaseAdmin
    .from("collection_batches")
    .update({
      batch_status: "finished",
      finished_at: new Date().toISOString(),
      total_items: rows.length,
      total_valid_items: rows.length,
      total_invalid_items: 0,
    })
    .eq("id", batch.id);

  return { batchId: batch.id, inserted: rows.length };
}

import { supabaseAdmin } from "../lib/supabaseAdmin";
import { normalizeTitle } from "../lib/normalize";

export async function runProcess(limit = 100) {
  const { data: rawItems, error } = await supabaseAdmin
    .from("raw_collected_items")
    .select("*")
    .eq("processing_status", "pending")
    .limit(limit);

  if (error) throw error;
  if (!rawItems?.length) return { processed: 0 };

  // carregar marcas para sugestão simples
  const { data: brands } = await supabaseAdmin.from("brands").select("id, name, slug");
  const brandRules = (brands ?? []).map((b) => ({
    id: b.id,
    token: b.name.toLowerCase(),
  }));

  let processedCount = 0;

  for (const raw of rawItems) {
    const normalized = normalizeTitle(raw.raw_title ?? "");

    let suggestedBrandId: number | null = null;
    let confidence = 0.3;

    for (const br of brandRules) {
      if (normalized.includes(br.token)) {
        suggestedBrandId = br.id;
        confidence = 0.8;
        break;
      }
    }

    const detectedIssues: any = {};
    if (!raw.image_url) detectedIssues.no_image = true;
    if (!raw.item_url) detectedIssues.no_url = true;
    if (!raw.raw_price || raw.raw_price <= 0) detectedIssues.bad_price = true;

    // duplicidade MVP: mesmo dedup_hash já processado?
    const { data: dup } = await supabaseAdmin
      .from("raw_collected_items")
      .select("id")
      .eq("dedup_hash", raw.dedup_hash)
      .neq("id", raw.id)
      .limit(1);

    const isDuplicate = !!(dup && dup.length);

    // grava processed
    const { data: processed, error: pErr } = await supabaseAdmin
      .from("processed_items")
      .insert({
        raw_item_id: raw.id,
        normalized_title: normalized || null,
        suggested_brand_id: suggestedBrandId,
        confidence_score: confidence,
        is_duplicate: isDuplicate,
        detected_issues: detectedIssues,
      })
      .select("*")
      .single();

    if (pErr) {
      await supabaseAdmin.from("raw_collected_items").update({ processing_status: "error" }).eq("id", raw.id);
      continue;
    }

    // prioridade
    let priority = 0;
    if (isDuplicate) priority += 50;
    if (confidence < 0.5) priority += 30;
    if (detectedIssues.no_image) priority += 10;

    await supabaseAdmin.from("triage_queue").insert({
      processed_item_id: processed.id,
      triage_status: "pending",
      priority,
    });

    await supabaseAdmin.from("raw_collected_items").update({ processing_status: "processed" }).eq("id", raw.id);

    processedCount++;
  }

  return { processed: processedCount };
}

import { supabaseAdmin } from "../lib/supabaseAdmin";

export async function approveTriageItem(params: {
  triageQueueId: number;
  product?: { main_name?: string; slug?: string; brand_id?: number; main_category_id?: number; main_image_url?: string };
  offer?: { affiliate_url?: string };
  notes?: string;
}) {
  // 1) carregar triage + processed + raw
  const { data: tq, error: tqErr } = await supabaseAdmin
    .from("triage_queue")
    .select(`
      id, triage_status,
      processed_items:processed_item_id (
        id, normalized_title, suggested_brand_id,
        raw_collected_items:raw_item_id (
          marketplace_id, external_item_id, raw_title, raw_price, raw_old_price, item_url, image_url
        )
      )
    `)
    .eq("id", params.triageQueueId)
    .single();

  if (tqErr) throw tqErr;
  if (!tq) throw new Error("triage item not found");

  const processed = (tq as any).processed_items;
  const raw = processed.raw_collected_items;

  // 2) upsert product (MVP: sempre cria)
  const mainName = params.product?.main_name ?? processed.normalized_title ?? raw.raw_title ?? "Produto";
  const slug = params.product?.slug ?? slugify(mainName);

  const { data: product, error: pErr } = await supabaseAdmin
    .from("products")
    .insert({
      main_name: mainName,
      slug,
      brand_id: params.product?.brand_id ?? processed.suggested_brand_id ?? null,
      main_category_id: params.product?.main_category_id ?? null,
      main_image_url: params.product?.main_image_url ?? raw.image_url ?? null,
      catalog_status: "published",
      updated_at: new Date().toISOString(),
    })
    .select("*")
    .single();

  if (pErr) throw pErr;

  // 3) criar offer
  const { data: offer, error: oErr } = await supabaseAdmin
    .from("offers")
    .insert({
      product_id: product.id,
      marketplace_id: raw.marketplace_id,
      external_item_id: raw.external_item_id,
      source_title: raw.raw_title,
      original_url: raw.item_url,
      affiliate_url: params.offer?.affiliate_url ?? null,
      current_price: raw.raw_price ?? 0,
      old_price: raw.raw_old_price ?? null,
      is_published: true,
      offer_status: "active",
      captured_at: new Date().toISOString(),
      updated_at: new Date().toISOString(),
    })
    .select("*")
    .single();

  if (oErr) throw oErr;

  // 4) atualizar fila + registrar ação
  await supabaseAdmin
    .from("triage_queue")
    .update({ triage_status: "approved", updated_at: new Date().toISOString() })
    .eq("id", tq.id);

  await supabaseAdmin.from("curation_actions").insert({
    triage_queue_id: tq.id,
    action_type: "approve",
    notes: params.notes ?? null,
  });

  return { product, offer };
}

function slugify(input: string) {
  return input
    .toLowerCase()
    .normalize("NFD").replace(/[\u0300-\u036f]/g, "")
    .replace(/[^a-z0-9]+/g, "-")
    .replace(/(^-|-$)/g, "");
}

export async function rejectTriageItem(params: {
  triageQueueId: number;
  rejection_reason_id?: number;
  notes?: string;
}) {
  await supabaseAdmin

    .update({ triage_status: "rejected", updated_at: new Date().toISOString() })

  await supabaseAdmin.from("curation_actions").insert({
    triage_queue_id: params.triageQueueId,
    action_type: "reject",
    rejection_reason_id: params.rejection_reason_id ?? null,
    notes: params.notes ?? null,
  });

  return { ok: true };
}

export async function api<T>(path: string, init?: RequestInit): Promise<T> {
  const res = await fetch(`/api${path}`, {
    headers: { "Content-Type": "application/json", ...(init?.headers ?? {}) },
    ...init,
  });

  if (!res.ok) {
    const msg = await res.text();
    throw new Error(msg || `HTTP ${res.status}`);
  }

  // para endpoints que redirecionam (click), não usar este wrapper
  return res.json() as Promise<T>;
}

import { useMutation, useQueryClient } from "@tanstack/react-query";
import { api } from "@/lib/api";

export function useApproveItem() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: (args: { id: number; notes?: string }) =>
      api(`/admin/triage/${args.id}/approve`, {
        method: "POST",
        body: JSON.stringify({ notes: args.notes }),
      }),
    onSuccess: () => qc.invalidateQueries({ queryKey: ["triage"] }),
  });
}

import { useTriageQueue } from "@/hooks/useTriageQueue";
import { useApproveItem } from "@/hooks/useApproveItem";
import { useRejectItem } from "@/hooks/useRejectItem";

export default function AdminTriagem() {
  const { data, isLoading, error } = useTriageQueue("pending");
  const approve = useApproveItem();
  const reject = useRejectItem();

  if (isLoading) return <div style={{ padding: 24 }}>Carregando fila…</div>;
  if (error) return <div style={{ padding: 24 }}>Erro: {(error as Error).message}</div>;

      <div style={{ display: "grid", gap: 12, marginTop: 16 }}>
        {(data ?? []).map((row: any) => {
          const p = row.processed_items;
          const raw = p?.raw_collected_items;

          return (
            <div key={row.id} style={{ border: "1px solid #ddd", borderRadius: 8, padding: 12 }}>
              <div style={{ display: "flex", gap: 12 }}>
                {raw?.image_url ? (
                  <img src={raw.image_url} alt="" style={{ width: 84, height: 84, objectFit: "cover", borderRadius: 6 }} />

                  <div style={{ width: 84, height: 84, background: "#eee", borderRadius: 6 }} />
                )}

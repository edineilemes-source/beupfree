🌐 1. HTTP CLIENT (ANTI-BLOQUEIO REAL)
mlHttpClient.ts
import fetch from "node-fetch";

const USER_AGENTS = [
  "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 Chrome/120 Safari/537.36",
  "Mozilla/5.0 (Macintosh; Intel Mac OS X 13_0) AppleWebKit/537.36 Chrome/119 Safari/537.36",
  "Mozilla/5.0 (X11; Linux x86_64) AppleWebKit/537.36 Chrome/118 Safari/537.36"
];

function getRandomUA() {
  return USER_AGENTS[Math.floor(Math.random() * USER_AGENTS.length)];
}

export async function fetchML(url: string, attempt = 1): Promise<string> {
  try {
    const res = await fetch(url, {
      headers: {
        "User-Agent": getRandomUA(),
        "Accept-Language": "pt-BR,pt;q=0.9",
        "Cache-Control": "no-cache",
        "Connection": "keep-alive"
      },
      timeout: 15000
    });

    if (!res.ok) throw new Error(`HTTP ${res.status}`);

    const html = await res.text();

    // 🔥 DETECÇÃO DE BLOQUEIO
    if (isBlocked(html)) {
      throw new Error("Blocked by Mercado Livre");
    }

    return html;

  } catch (err) {
    if (attempt <= 3) {
      console.log(`Retry ${attempt} → ${url}`);
      await sleep(1000 * attempt);
      return fetchML(url, attempt + 1);
    }

    throw err;
  }
}

function isBlocked(html: string) {
  return (
    html.includes("Access Denied") ||
    html.includes("Pardon Our Interruption") ||
    html.length < 5000
  );
}

function sleep(ms: number) {
  return new Promise(res => setTimeout(res, ms));
}

import { load } from "cheerio";

export function parseMLPage(html: string) {
  const $ = load(html);
  const items: any[] = [];

  $(".ui-search-result").each((_, el) => {
    try {
      const title = $(el).find("h2").text().trim();

      let item_url = $(el).find("a").attr("href") || "";
      item_url = item_url.split("?")[0];

      const image_url =
        $(el).find("img").attr("data-src") ||
        $(el).find("img").attr("src");

      const price = $(el).find(".price-tag-fraction").first().text();
      const current_price = parseInt(price.replace(/\./g, ""));

      const discountText = $(el)
        .find(".ui-search-price__discount")
        .text();

      const discount_percent = parseInt(
        discountText.replace(/[^\d]/g, "")
      );

      const external_item_id = extractMLB(item_url);

      if (!title || !item_url || !current_price) return;

      items.push({
        external_item_id,
        title,
        item_url,
        image_url,
        current_price,
        discount_percent,
        payload_json: { raw_discount: discountText }
      });

    } catch (err) {}
  });

  return items;
}

function extractMLB(url: string) {
  const match = url.match(/MLB-\d+/);
  return match ? match[0] : null;
}

mlPaginator.ts
import { fetchML } from "./mlHttpClient";
import { parseMLPage } from "./mlParser";

export async function paginateML(baseUrl: string) {
  const results: any[] = [];
  let previousPageHash = "";

  for (let offset = 1; offset <= 2000; offset += 48) {
    const url =
      offset === 1
        ? baseUrl
        : `${baseUrl}&_Desde=${offset}`;

    console.log("Página:", offset);

    const html = await fetchML(url);
    const items = parseMLPage(html);

    if (!items.length) break;

    // 🔥 DETECTAR REPETIÇÃO (anti-loop infinito)
    const currentHash = JSON.stringify(items.slice(0, 5));
    if (currentHash === previousPageHash) {
      console.log("Página repetida → stop");
      break;
    }

    previousPageHash = currentHash;

    results.push(...items);

    await sleep(random(800, 1800));
  }

  return deduplicate(results);
}

function deduplicate(items: any[]) {
  const map = new Map();

  for (const item of items) {
    const key = item.external_item_id || item.item_url;
    if (!map.has(key)) map.set(key, item);
  }

  return Array.from(map.values());
}

function sleep(ms: number) {
  return new Promise(res => setTimeout(res, ms));
}

function random(min: number, max: number) {
  return Math.floor(Math.random() * (max - min) + min);
}

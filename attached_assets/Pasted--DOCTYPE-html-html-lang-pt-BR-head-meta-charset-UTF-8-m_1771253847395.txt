  <script>
    // ✅ TROQUE AQUI quando mudar o túnel
    const API_BASE = "https://howard-positioning-indirect-consortium.trycloudflare.com";

    const $ = (id) => document.getElementById(id);

    function formatBRL(n){
      if (n === null || n === undefined) return "—";
      return Number(n).toLocaleString("pt-BR",{style:"currency",currency:"BRL"});
    }

    function escapeHtml(s){
      return String(s ?? "")
        .replaceAll("&","&amp;")
        .replaceAll("<","&lt;")
        .replaceAll(">","&gt;")
        .replaceAll('"',"&quot;")
        .replaceAll("'","&#039;");
    }

    function buildCard(p){
      const nome = escapeHtml(p.nome || "Produto");
      const marca = escapeHtml(p.marca || "");
      const preco = formatBRL(p.preco_atual);
      const original = p.preco_original ? formatBRL(p.preco_original) : "";
      const desconto = (p.desconto_percent !== null && p.desconto_percent !== undefined)
        ? `${p.desconto_percent}% OFF`
        : "";
      const link = p.link_afiliado || p.url;

      const img = (p.imagens && p.imagens[0]) ? p.imagens[0] : "";
      const rating = p.avaliacao_media ? `⭐ ${p.avaliacao_media}` : "";
      const qtd = p.qtd_avaliacoes ? `(${p.qtd_avaliacoes})` : "";

      return `
        <article class="card">
          ${img ? `<img class="img" src="${img}" alt="${nome}">` : `<div class="img"></div>`}
          <div class="title">${nome}</div>
          <div class="muted">${marca}</div>

          <div class="price">${preco}</div>
          ${original ? `<div class="muted"><s>${original}</s> ${desconto ? " • " + desconto : ""}</div>` : (desconto ? `<div class="muted">${desconto}</div>` : "")}

          ${(rating || qtd) ? `<div class="muted" style="margin-top:6px">${rating} ${qtd}</div>` : ""}

          <div style="margin-top:12px">
            <a class="btn btn-primary" href="${link}" target="_blank" rel="noopener">
              Conferir no Mercado Livre ↗
            </a>
          </div>
        </article>
      `;
    }

    async function carregar(){
      $("error").style.display = "none";
      $("loading").style.display = "block";
      $("status").textContent = "Carregando…";
      $("grid").innerHTML = "";

      const q = $("q").value.trim() || "tenis";
      const enrich = $("enrich").value;

      // Seu backend atual está fixo em /tenis.
      // Então aqui vamos chamar /triagem normalmente.
      // Depois, se você quiser busca dinâmica, eu ajusto sua API para aceitar ?q=
      const url = `${API_BASE}/triagem?enrich=${enrich}`;

      try{
        const res = await fetch(url, { method:"GET" });
        if(!res.ok) throw new Error("Falha HTTP " + res.status);

        const data = await res.json();

        $("meta").textContent =
          `Fonte: ${data.fonte} • Coletados: ${data.coletados} • Entregues: ${data.entregues}`;

        $("status").textContent = "Ok ✅";

        const produtos = Array.isArray(data.produtos) ? data.produtos : [];
        if(!produtos.length){
          $("grid").innerHTML = `<div class="muted">Nenhum produto retornou.</div>`;
          return;
        }

        $("grid").innerHTML = produtos.map(buildCard).join("");
      }catch(e){
        $("status").textContent = "Erro ❌";
        $("error").style.display = "block";
        $("error").textContent = "Erro ao carregar triagem: " + (e.message || e);
      }finally{
        $("loading").style.display = "none";
      }
    }

    $("btnBuscar").addEventListener("click", carregar);
    $("btnAtualizar").addEventListener("click", carregar);

    // carrega automaticamente ao abrir
    carregar();
  </script>
